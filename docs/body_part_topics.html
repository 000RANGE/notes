Text Pre-processing¶

Steps:

  1. remove newline characters,
  2. transfer to lower case,
  3. remove punctuations and numbers,
  4. tokenization,
  5. remove stopwords,
  6. lemmatization
In [34]:
processed_data = []
for document in notes:
    # Remove newline characters & transfer to lowercase
    document = document.lower()
    # remove punctuations and numbers
    translator =str.maketrans('','',string.punctuation+'0123456789')
    document = document.translate(translator)
    
    # Tokenize
    word_list = word_tokenize(document)
    
    # Remove stop words
    stop_words = stopwords.words('english')
    filtered_word_list = [i for i in word_list if i not in stop_words]
      
    # Lemmatization to reduce features
    lemmatizer = WordNetLemmatizer()
    lemmatized_words = [lemmatizer.lemmatize(i) for i in filtered_word_list]
    processed_data.append(lemmatized_words)

def assign_topics(topic_list, lda_model,corpus):
    lda_topic_assignment = [max(p,key=lambda item: item[1]) for p in lda_model[corpus]]
    topic_label = [lda_topic_assignment[i][0] for i in range(len(lda_topic_assignment))]
    topics = []
    for label in topic_label:
        if label in topic_list:
            topics.append(1)
        else:
            topics.append(0)
    return topics
topics = assign_topics(topic_list, lda_model,corpus)

# accuracy
def accuracy(prediction, label):
    count = 0
    for i in range(len(label)):
        if prediction[i] == label[i]:
            count += 1
    return count / len(prediction)

# Precision TP / (TP + FP)
def precision(prediction, label):
    TP = 0
    FP = 0
    for i in range(len(label)):
        if prediction[i] == 1 and label[i] == 1:
            TP += 1
        elif prediction[i] == 1 and label[i] == 0:
            FP += 1
    return TP / (TP + FP)
precision = precision(df_new.prediction, df_new.label)

# Recall TP / (TP + FN)
def recall(prediction, label):
    TP = 0
    FN = 0
    for i in range(len(label)):
        if prediction[i] == 1 and label[i] == 1:
            TP += 1
        elif prediction[i] == 0 and label[i] == 1:
            FN += 1
    return TP / (TP + FN)
recall = recall(df_new.prediction, df_new.label)
